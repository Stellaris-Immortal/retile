# Restaurant Megachain
building_restaurant_megachain = {
	base_buildtime = 360
	base_cap_amount = 1
	category = trade
	can_be_disabled = no
	
	potential = {
		exists = owner
		owner = { 
			is_regular_empire = yes
			has_tradition = tr_prosperity_public_works
		}
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = penal_colony }
	}
	
	allow = {
		OR = { 
			retile_urban = yes
			has_modifier = resort_colony
		}
		#retile_ready = yes
	}

	destroy_trigger = {
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = @cost_med
		}	
		upkeep = {
			minerals = 0
			energy = 10
		}		
	}
	
	planet_modifier = {
		pop_growth_speed = 0.10
		planet_housing_add = 2
		job_custodian_add = 2
	}
	
	triggered_desc = {
		trigger = { exists = owner }
		text = retile_restaurant_megachain_effect
	}
	
	ai_weight = {
		weight = 0
		
		modifier = {
			weight = 10000
			owner = { num_owned_planets < 5 }
			count_pops = {
				limit = {  
					has_job = job_clerk
				}
				count > 4
			}
		}
		modifier = {	# Help solve energy problems!!!
			weight = 200000
			owner = { has_monthly_income = { resource = energy value < 25 } }
		}
		modifier = {	# goods security
			weight = 20000
			owner = { has_monthly_income = { resource = consumer_goods value < 25 } }
		}
	}
}

building_metascience_lab_1 = {
	base_buildtime = 360
	base_cap_amount = 1
	category = research
	can_be_disabled = no
	
	potential = {
		NOT = { has_modifier = resort_colony }	
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = penal_colony }
		owner = { has_tradition = tr_harmony_bulwark_of_harmony }
	}
	
	allow = {
		retile_urban = yes
		#retile_ready = yes
	}
	
	convert_to = {
		building_research_lab_1
	}

	destroy_trigger = {
		OR = {
			retile_urban = no
			has_modifier = slave_colony
			has_modifier = resort_colony
			has_modifier = penal_colony
			NOT = { owner = { has_tradition = tr_harmony_bulwark_of_harmony } }
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @cost_low
		}
		upkeep = {
			minerals = 0
			energy = 10
		}
		upkeep = {
			trigger = { owner = { is_hive_empire = yes } }
			energy = -5
			consumer_goods = 1
		}
	}
	
	upgrades = {
		"building_metascience_lab_2"
	}
	
	prerequisites = {
		"tech_basic_science_lab_1"
	}
	
	show_tech_unlock_if = {
		has_tradition = tr_harmony_bulwark_of_harmony
	}
	
	planet_modifier = {
		pop_government_ethic_attraction = 0.15
	}
	country_modifier = {
		country_resource_max_add = 1000
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			planet_housing_add = 1
			job_custodian_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			planet_housing_add = 1
			job_maintenance_drone_add = 1
		}
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
		}
		text = retile_metascience_lab_1_effect
	}

	ai_weight = {
		weight = 200
		
		modifier = {	# Safe to advance our empire?
			weight = 300
			retile_advanced_economy = yes
		}
		modifier = {
			weight = 150000
			is_capital = yes
		}
		modifier = {	# behind in tech!!!
			weight = 100
			owner = {
				any_neighbor_country = {
					is_country_type = default
					prev = {
						relative_power = {
							who = prev
							category = technology 
							value > inferior
						}
						OR = {
							NOT = { relative_power = {
								who = prev 
								category = all
								value = overwhelming
							} }
							NOT = { relative_power = {
								who = prev 
								category = all
								value < equivalent
							} }
						}
					}
				}
			}
		}
		modifier = {	# Ministry!!!
			factor = 5
			OR = {
				has_building = building_institute
				#has_building = building_supercomputer
			}
		}
		modifier = {	# consumer goods collapse!!!
			factor = -1
			owner = { has_monthly_income = { resource = consumer_goods value < -50 } }
		}
	}
	
	ai_resource_production = {
		physics_research = 1
		society_research = 1
		engineering_research = 1
		trigger = {
			always = yes
		}
	}
}
