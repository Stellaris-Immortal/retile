@cost_low = 500
@cost_med = 1000
@cost_high = 1500
@cost_ultra = 2000

#############
# Trade buildings
#############
building_commercial_zone = {
	base_buildtime = 360
	base_cap_amount = 1
	category = trade
	can_be_disabled = no
	
	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
	}
	
	allow = {
		OR = { 
			retile_urban = yes
			has_modifier = resort_colony
		}
		#retile_ready = yes
	}

	destroy_trigger = {
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			NOR = { 
				retile_urban = yes
				has_modifier = resort_colony
			}
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = @cost_low
		}	
		upkeep = {
			minerals = 0
			energy = 10
		}		
	}
	
	prerequisites = {
		"tech_interplanetary_commerce"
	}
	
	show_tech_unlock_if = { 
		is_regular_empire = yes
	}
	
	country_modifier = {
		country_resource_max_add = 1000
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			planet_housing_add = 1
			job_custodian_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			planet_housing_add = 1
			job_maintenance_drone_add = 1
		}
	}
	
	triggered_desc = {
		trigger = { exists = owner }
		text = retile_commercial_zone_effect
	}
	
	upgrades = {
		"building_commercial_megaplex"
	}
	
	ai_weight = {
		weight = 0
		
		modifier = {	# Help solve energy problems!!!
			weight = 200000
			OR = {
				is_capital = yes
				owner = { has_monthly_income = { resource = energy value < 25 } }
			}
		}
		modifier = {	# goods security
			weight = 20000
			owner = { has_monthly_income = { resource = consumer_goods value < 25 } }
		}
		modifier = {	# Stock Exchange!!! (rarely relevant - ai should only build this one first...
			factor = 5
			has_building = building_galactic_stock_exchange
		}
	}
}
