totalitarian = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_authoritarian
	
	unique = yes
	support_multiplier = 1
	
	resources = {
		category = pop_factions
		produces = { 
			trigger = { owner = { has_valid_civic = civic_pacifist } }
			unity = 10
		}
		produces = { 
			trigger = { owner = { NOT = { has_valid_civic = civic_pacifist } } }
			influence = 2
		}
	}

	is_potential = {
		OR = {
			years_passed > 20
			has_authority = auth_oligarchic
			has_authority = auth_dictatorial
			has_authority = auth_imperial
			has_ethic = ethic_authoritarian
			has_ethic = ethic_fanatic_authoritarian
		}
		NOR = { 
			has_ethic = ethic_gestalt_consciousness
			AND = {
				has_valid_civic = civic_inwards_perfection
				NOR = {
					has_ethic = ethic_authoritarian
					has_ethic = ethic_fanatic_authoritarian
				}
			}
		}
	}
	
	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}
	
	can_join_pre_triggers = {
		is_being_purged = no
		is_being_assimilated = no
	}
	
	can_join_faction = {
		is_shackled_robot = no
		has_culture_shock = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			has_ethic = ethic_authoritarian
			has_ethic = ethic_fanatic_authoritarian
		}
		OR = {
			is_enslaved = no
			has_slavery_type = { type = slavery_domestic }
		}
	}
	
	attraction = {
		base = 100
	}
	
	leader = {
		base = 100
		
		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}

	# Wants State Police Policy
	demand = { 
		title = "AUTHORITARIAN_STATE_POLICE"
		unfulfilled_title = "AUTHORITARIAN_NO_STATE_POLICE"
		desc = "AUTHORITARIAN_STATE_POLICE_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = {exists = owner }

		trigger = {
			owner = { has_policy_flag = criminal_state_police }
		}
	}
	
	# Wants an Oligarchy
	demand = {
		title = "AUTHORITARIAN_OLIGARCHY"
		unfulfilled_title = "AUTHORITARIAN_NO_OLIGARCHY"
		desc = "AUTHORITARIAN_OLIGARCHY_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = { exists = owner }

		trigger = {
			owner = { has_authority = auth_oligarchic }
		}
	}
	
	# Wants Priority Healthcare Policy
	demand = { 
		title = "AUTHORITARIAN_PRIORITY_HEALTHCARE"
		unfulfilled_title = "AUTHORITARIAN_NO_PRIORITY_HEALTHCARE"
		desc = "AUTHORITARIAN_PRIORITY_HEALTHCARE_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = { exists = owner }

		trigger = {
			owner = { has_policy_flag = healthcare_for_profit }
		}
	}

	# Wants Priority Education Policy
	demand = {
		title = "AUTHORITARIAN_PRIORITY_EDUCATION"
		unfulfilled_title = "AUTHORITARIAN_NO_PRIORITY_EDUCATION"
		desc = "AUTHORITARIAN_PRIORITY_EDUCATION_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = { exists = owner }

		trigger = {
			owner = { has_policy_flag = education_for_profit }
		}
	}
	
	# Wants Workers have limited rights
	demand = { 
		title = "AUTHORITARIAN_LIMITED_WORKER_RIGHTS"
		unfulfilled_title = "AUTHORITARIAN_NO_LIMITED_WORKER_RIGHTS"
		desc = "AUTHORITARIAN_LIMITED_WORKER_RIGHTS_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = { exists = owner }

		trigger = {
			owner = { 
				OR = {
					has_policy_flag = slavery_secretly_allowed 
					has_policy_flag = slavery_allowed 
				}
			}
		}
	}
	
	demand = { 
		title = "AUTHORITARIAN_PROSPERITY"
		unfulfilled_title = "AUTHORITARIAN_NO_PROSPERITY"
		desc = "AUTHORITARIAN_PROSPERITY_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = { exists = owner }

		trigger = {
			owner = {
				has_tradition = tr_prosperity_adopt
			}
		}
	}
	
	demand = {
		title = "AUTHORITARIAN_STRATIFICATION"
		unfulfilled_title = "AUTHORITARIAN_NO_STRATIFICATION"
		desc = "AUTHORITARIAN_STRATIFICATION_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				NOT = {
					any_owned_pop = {
						OR = {
							has_living_standard = { type = living_standard_shared_burden }
							has_living_standard = { type = living_standard_utopian }
							has_living_standard = { type = living_standard_academic_privilege } 
							has_living_standard = { type = living_standard_good }
						}
					}				
				}
			}
		}
	}
	
	demand = {
		title = "AUTHORITARIAN_CAPACITY_BOOSTERS"
		unfulfilled_title = "AUTHORITARIAN_NO_CAPACITY_BOOSTERS"
		desc = "AUTHORITARIAN_CAPACITY_BOOSTERS_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = {
			exists = owner
			owner = {
				OR = {
					has_technology = tech_selected_lineages
					has_technology = tech_capacity_boosters
				}
			}
		}

		trigger = {
			owner = {
				OR = {
					has_policy_flag = leader_enhancement_selected_lineages
					has_policy_flag = leader_enhancement_capacity_boosters
				}
			}
		}
	}
	
	demand = {
		title = "COMMON_DICTATOR"
		unfulfilled_title = "COMMON_DICTATOR"
		desc = "COMMON_DICTATOR_DESC"

		fulfilled_effect = -10
		unfulfilled_effect = 0

		potential = { 
			exists = owner
			owner = {
				has_authority = auth_dictatorship
			}
		}

		trigger = {
			owner = {
				has_authority = auth_dictatorship
			}
		}
	}
	
	demand = {
		title = "COMMON_EMPEROR"
		unfulfilled_title = "COMMON_EMPEROR"
		desc = "COMMON_EMPEROR_DESC"

		fulfilled_effect = -20
		unfulfilled_effect = 0

		potential = { 
			exists = owner 
			owner = {
				has_authority = auth_imperial
			}
		}

		trigger = {
			owner = {
				has_authority = auth_imperial
			}
		}
	}
	
	# OUTRAGE
	demand = {
		title = "FACTION_OUTRAGED_GC"
		unfulfilled_title = "FACTION_OUTRAGED_GC"
		desc = "FACTION_OUTRAGED_GC_DESC"

		fulfilled_effect = -10
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				has_country_flag = faction_authoritarian_outraged_gc
			}
		}

		trigger = {
			owner = {
				has_country_flag = faction_authoritarian_outraged_gc
			}
		}
	}
	
	on_create = {
		pop_faction_event = { id = factions.1 }
	}

	on_destroy = {
		if = {
			limit = {
				has_modifier = promoted_authoritarian
			}
			remove_modifier = promoted_authoritarian
		}
		if = {
			limit = {
				has_modifier = suppressed_authoritarian
			}
			remove_modifier = suppressed_authoritarian
		}
	}

	actions = {
		embrace_faction = {
			title = "EMBRACE_FACTION"
			description = "EMBRACE_FACTION_DESC"
			
			cost = {
				influence = 250
			}
			
			potential = {
				exists = owner
				owner = {
					OR = {
						is_subject = no
						NOT = { is_subject_type = dominion }
					}
				}
			}
			
			valid = {
				custom_tooltip = {
					fail_text = EMBRACE_FACTION_COOLDOWN
					parameter:empire = {
						NOT = { has_modifier = embraced_faction_timer }
					}
				}
				support > 0.20
				parameter:empire = {
					NOT = { has_ethic = "ethic_fanatic_authoritarian" }
				}
			}

			effect = {
				add_modifier = { modifier = embraced_faction days = 3600 }
				parameter:empire = {
					shift_ethic = ethic_authoritarian
					hidden_effect = {
						add_modifier = { modifier = embraced_faction_timer days = 3600 }
						every_pop_faction = {
							limit = { NOT = { is_same_value = root } }
							add_modifier = { modifier = embraced_another_faction days = 3600 }
						}
					}
				}
				hidden_effect = {
					save_event_target_as = TargetFaction
					parameter:empire = {
						every_relation = {
							limit = {
								is_ai = no
								is_country_type = default			
								has_communications = prev
							}
							country_event = { id = factions.2000 }
						}
					}				
				}				
			}
			
			ai_weight = {
				base = 1
				modifier = {
					factor = 0
					support < 0.50
					owner = {
						has_ethic = ethic_authoritarian
					}
				}
				modifier = {
					factor = 0
					owner = { has_valid_civic = civic_fanatic_purifiers }
				}				
				modifier = {
					factor = 0
					NOT = { 
						owner = {
							exists = ruler
							ruler = {
								leader_of_faction = totalitarian
							}
						}
					}
				}
				modifier = {
					factor = 0
					owner = {
						count_pop_factions = {
							count < 4
						}					
					}
				}
			}
		}
		
		promote_faction = {
			title = "PROMOTE_FACTION"
			description = "PROMOTE_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_authoritarian
						has_modifier = promoted_authoritarian
						has_valid_civic = civic_anarchist
					}
				}
			}

			effect = {
				parameter:empire = {
					add_modifier = { modifier = promoted_authoritarian days = -1 }
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_promote_faction = {
			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = promoted_authoritarian
				}
			}

			effect = {
				parameter:empire = {
					remove_modifier = promoted_authoritarian
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_authoritarian
						has_modifier = promoted_authoritarian
						has_valid_civic = civic_anarchist
					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }
				parameter:empire = {
					add_modifier = { modifier = suppressed_authoritarian days = -1 }
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = suppressed_authoritarian
				}
			}

			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_authoritarian
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
	}
}
