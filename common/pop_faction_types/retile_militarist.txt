##############################
# Imperialist Faction
##############################

imperialist = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_militarist
	
	unique = yes
	support_multiplier = 1
	
	resources = {
		category = pop_factions
		produces = { influence = 2 }
	}

	is_potential = {
		OR = {
			years_passed > 20
			has_country_flag = first_contact_event
			has_ethic = ethic_militarist
			has_ethic = ethic_fanatic_militarist
		}
		NOR = { 
			has_ethic = ethic_gestalt_consciousness
			has_valid_civic = civic_pacifist
			has_valid_civic = civic_inwards_perfection
		}
	}
	
	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}
	
	can_join_pre_triggers = {
		is_being_purged = no
		is_being_assimilated = no
	}

	can_join_faction = {
		is_shackled_robot = no
		has_culture_shock = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			has_ethic = ethic_militarist
			has_ethic = ethic_fanatic_militarist
		}
		OR = {
			is_enslaved = no
			has_slavery_type = { type = slavery_domestic }
		}
	}
	
	attraction = {
		base = 100
	}
	
	leader = {
		base = 100
		
		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}
	
	# Loves Slaying
	demand = {
		title = "MILITARIST_SLAYER"
		unfulfilled_title = "MILITARIST_NO_SLAYER"
		desc = "MILITARIST_SLAYER_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = 0

		potential = {
			exists = owner
			owner = {
				host_has_dlc = "Leviathans Story Pack"
				any_country = {
					OR = {
						is_country_type = guardian
						is_country_type = guardian_dragon
						is_country_type = guardian_stellarite
						is_country_type = guardian_wraith #
						is_country_type = guardian_hiver
						is_country_type = guardian_horror
						is_country_type = guardian_fortress
						is_country_type = guardian_dreadnought
						is_country_type = guardian_sphere
						is_country_type = guardian_scavenger_bot
						is_country_type = guardian_elderly_tiyanki
						is_country_type = guardian_hatchling
					}
					has_communications = prev
				}
			}
		}

		trigger = {
			owner = {
				has_country_flag = killed_guardian
			}
		}
	}
	
	demand = {
		title = "MILITARIST_AGG_STANCE"
		unfulfilled_title = "MILITARIST_NO_AGG_STANCE"
		desc = "MILITARIST_AGG_STANCE_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				OR = {
					has_policy_flag = diplo_stance_belligerent
					has_policy_flag = diplo_stance_supremacist
				}
			}
		}
	}
	
	# Wants War
	demand = { 
		title = "MILITARIST_WAR"
		unfulfilled_title = "MILITARIST_NO_WAR"
		desc = "MILITARIST_WAR_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = { 
			exists = owner
			owner = { has_encountered_other_species = yes }
		}

		trigger = {
			owner = { is_at_war = yes }
		}
	}
	
	# Wants Supremacy
	demand = {
		title = "MILITARIST_SUPREMACY"
		unfulfilled_title = "MILITARIST_NO_SUPREMACY"
		desc = "MILITARIST_SUPREMACY_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = { 
			exists = owner
		}

		trigger = {
			owner = { has_tradition = tr_supremacy_adopt }
		}
	}
	
	# Wants Military Police
	demand = {
		title = "MILITARIST_MILITARY_POLICE"
		unfulfilled_title = "MILITARIST_NO_MILITARY_POLICE"
		desc = "MILITARIST_MILITARY_POLICE_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = { exists = owner }

		trigger = {
			owner = { has_policy_flag = criminal_military_police }
		}
	}
	
	# Wants Indescriminate Bombardment
	demand = {
		title = "MILITARIST_BOMBARDMENT"
		unfulfilled_title = "MILITARIST_NO_BOMBARDMENT"
		desc = "MILITARIST_BOMBARDMENT_DESC"
		
		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = { exists = owner }

		trigger = {
			owner = { 
				OR = {
					has_policy_flag = orbital_bombardment_indiscriminate
					has_policy_flag = orbital_bombardment_armageddon
				}
			}
		}
	}
	
	# Wants Unrestricted Wars
	demand = {
		title = "MILITARIST_UNRESTRICTED_WAR"
		unfulfilled_title = "MILITARIST_NO_UNRESTRICTED_WAR"
		desc = "MILITARIST_UNRESTRICTED_WAR_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = { exists = owner }

		trigger = {
			owner = { has_policy_flag = unrestricted_wars }
		}
	}
	
	# Hates Being Weak
	demand = { 
		title = "MILITARIST_WEAK"
		unfulfilled_title = "MILITARIST_NO_WEAK"
		desc = "MILITARIST_WEAK_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = { 
			exists = owner
			owner = { has_encountered_other_species = yes }
		}

		trigger = {
			NOT = {
				owner = {
					any_country = {
						is_country_type = default
						has_communications = root.owner
						relative_power = { who = root.owner value >= superior }
					}
				}
			}
		}
	}

	demand = {
		title = "COMMON_DICTATOR"
		unfulfilled_title = "COMMON_DICTATOR"
		desc = "COMMON_DICTATOR_DESC"

		fulfilled_effect = -10
		unfulfilled_effect = 0

		potential = { 
			exists = owner
			owner = {
				has_auth = auth_dictatorship
			}
		}

		trigger = {
			owner = {
				has_auth = auth_dictatorship
			}
		}
	}
	
	demand = {
		title = "COMMON_EMPEROR"
		unfulfilled_title = "COMMON_EMPEROR"
		desc = "COMMON_EMPEROR_DESC"

		fulfilled_effect = -20
		unfulfilled_effect = 0

		potential = { 
			exists = owner 
			owner = {
				has_auth = auth_imperial
			}
		}

		trigger = {
			owner = {
				has_auth = auth_imperial
			}
		}
	}
	
	# OUTRAGE
	demand = {
		title = "FACTION_OUTRAGED_GC"
		unfulfilled_title = "FACTION_OUTRAGED_GC"
		desc = "FACTION_OUTRAGED_GC_DESC"

		fulfilled_effect = -10
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				has_country_flag = faction_militarist_outraged_gc
			}
		}

		trigger = {
			owner = {
				has_country_flag = faction_militarist_outraged_gc
			}
		}
	}
	
	on_create = {
		pop_faction_event = { id = factions.1 }
	}

	on_destroy = {
		if = {
			limit = {
				has_modifier = promoted_militarist
			}
			remove_modifier = promoted_militarist
		}
		if = {
			limit = {
				has_modifier = suppressed_militarist
			}
			remove_modifier = suppressed_militarist
		}
	}
	
	actions = {
		embrace_faction = {
			title = "EMBRACE_FACTION"
			description = "EMBRACE_FACTION_DESC"
			
			cost = {
				influence = 250
			}
			
			potential = {
				exists = owner
				owner = {
					OR = {
						is_subject = no
						NOT = { is_subject_type = dominion }
					}
				}
			}
			
			valid = {
				custom_tooltip = {
					fail_text = EMBRACE_FACTION_COOLDOWN
					parameter:empire = {
						NOT = { has_modifier = embraced_faction_timer }
					}
				}
				support > 0.20
				parameter:empire = {
					NOT = { has_ethic = "ethic_fanatic_militarist" }
				}
			}

			effect = {
				add_modifier = { modifier = embraced_faction days = 3600 }
				parameter:empire = {
					shift_ethic = ethic_militarist
					hidden_effect = {
						add_modifier = { modifier = embraced_faction_timer days = 3600 }
						every_pop_faction = {
							limit = { NOT = { is_same_value = root } }
							add_modifier = { modifier = embraced_another_faction days = 3600 }
						}
					}
				}
				hidden_effect = {
					save_event_target_as = TargetFaction
					parameter:empire = {
						every_relation = {
							limit = {
								is_ai = no
								is_country_type = default			
								has_communications = prev
							}
							country_event = { id = factions.2000 }
						}
					}				
				}				
			}
			
			ai_weight = {
				base = 1
				modifier = {
					factor = 0
					support < 0.50
					owner = {
						has_ethic = ethic_militarist
					}
				}
				modifier = {
					factor = 0
					owner = { has_valid_civic = civic_fanatic_purifiers }
				}
				modifier = {
					factor = 0
					NOT = { 
						owner = {	
							exists = ruler
							ruler = {
								leader_of_faction = imperialist
							}
						}
					}
				}
				modifier = {
					factor = 0
					owner = {
						count_pop_factions = {
							count < 4
						}					
					}
				}				
			}
		}
		promote_faction = {
			title = "PROMOTE_FACTION"
			description = "PROMOTE_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_militarist
						has_modifier = promoted_militarist
						has_valid_civic = civic_anarchist
					}
				}
			}

			effect = {
				parameter:empire = {
					add_modifier = { modifier = promoted_militarist days = -1 }
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_promote_faction = {
			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = { has_modifier = promoted_militarist }
			}

			effect = {
				parameter:empire = {
					remove_modifier = promoted_militarist
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_militarist
						has_modifier = promoted_militarist
						has_valid_civic = civic_anarchist
					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }
				parameter:empire = {
					add_modifier = { modifier = suppressed_militarist days = -1 }
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = { has_modifier = suppressed_militarist }
			}

			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_militarist
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
	}
}
