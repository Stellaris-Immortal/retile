# Isolationist Faction

isolationist = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_xenophobe
	
	unique = yes
	support_multiplier = 1
	
	resources = {
		category = pop_factions
		produces = { 
			trigger = { owner = { has_valid_civic = civic_pacifist } }
			unity = 10
		}
		produces = { 
			trigger = { owner = { NOT = { has_valid_civic = civic_pacifist } } }
			influence = 2
		}		
	}

	is_potential = {
		OR = {
			years_passed > 20
			has_country_flag = amoeba_encountered
			has_country_flag = crystal_encountered
			has_country_flag = drone_encountered
			has_country_flag = cloud_encountered
			has_country_flag = tiyanki_encountered
			has_country_flag = has_encountered_other_empire
			has_valid_civic = civic_pacifist
		}
		NOT = { has_ethic = ethic_gestalt_consciousness }
	}
	
	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_being_purged = no
		is_being_assimilated = no
	}

	can_join_faction = {
		is_shackled_robot = no
		has_culture_shock = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			has_ethic = ethic_xenophobe
			has_ethic = ethic_fanatic_xenophobe
		}
		OR = {
			is_enslaved = no
			has_slavery_type = { type = slavery_domestic }
		}
	}
	
	attraction = {
		base = 100
		
		modifier = {
			factor = 1.25
			OR = { 
				has_job = bureaucrat_head
				has_job = bureaucrat
			}
		}
		
		modifier = {
			factor = 1.5
			is_same_species = owner_species
		}
		
		modifier = {
			factor = 1.25
			is_pop_category = ruler
		}
		
		modifier = {
			factor = 0.75
			has_citizenship_rights = no
		}
		
		modifier = {
			factor = 0.5
			NOT = { is_same_species = owner_species }
		}
	}
	
	leader = {
		base = 100
		
		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}
	
	# Wants Rival
	demand = { 
		title = "XENOPHOBE_RIVAL"
		unfulfilled_title = "XENOPHOBE_NO_RIVAL"
		desc = "XENOPHOBE_RIVAL_DESC"

		fulfilled_effect = 20
		unfulfilled_effect = 0

		potential = { 
			exists = owner
			owner = { has_encountered_other_species = yes }
		}

		trigger = {
			any_country = {
				prev.owner = { is_rival = PREV }
			}
		}
	}
	
	# Wants Purity
	demand = {
		title = "XENOPHOBE_PURITY"
		unfulfilled_title = "XENOPHOBE_NO_PURITY"
		desc = "XENOPHOBE_PURITY_DESC"
		
		fulfilled_effect = 10
		unfulfilled_effect = 0

		potential = {
			exists = owner 
			owner = { has_encountered_other_species = yes }
		}

		trigger = {
			NOT = {
				owner = { 
					any_owned_pop = {
						NOT = { is_same_species = owner_species }
						is_robot_pop = no
						is_sapient = yes
						is_being_purged = no
						is_enslaved = no
					}
				}
			}
		}
	}
	
	# Wants Bureaucracy
	demand = {
		title = "XENOPHOBE_ISOLATIONISM"
		unfulfilled_title = "XENOPHOBE_NO_ISOLATIONISM"
		desc = "XENOPHOBE_ISOLATIONISM_DESC"
		
		fulfilled_effect = 20
		unfulfilled_effect = 0

		potential = {
			exists = owner
			owner = { has_technology = tech_xenodiplomacy }
		}

		trigger = {
			owner = { has_policy_flag = diplo_stance_isolationist }
		}
	}
	
	# Wants Closed Borders
	demand = {
		title = "XENOPHOBE_CLOSED_BORDERS"
		unfulfilled_title = "XENOPHOBE_NO_CLOSED_BORDERS"
		desc = "XENOPHOBE_CLOSED_BORDERS_DESC"
		
		fulfilled_effect = 20
		unfulfilled_effect = 0

		potential = { exists = owner }

		trigger = {
			owner = { has_policy_flag = border_policy_closed }
		}
	}
	
	# Wants Pop Controls
	demand = {
		title = "XENOPHOBE_POP_CONTROLS"
		unfulfilled_title = "XENOPHOBE_NO_POP_CONTROLS"
		desc = "XENOPHOBE_POP_CONTROLS_DESC"
		
		fulfilled_effect = 10
		unfulfilled_effect = 0

		potential = { exists = owner }

		trigger = {
			owner = { has_policy_flag = population_controls_allowed }
		}
	}
	
	# Wants Purge Presapients
	demand = {
		title = "XENOPHOBE_PRESAPIENTS_PURGE"
		unfulfilled_title = "XENOPHOBE_NO_PRESAPIENTS_PURGE"
		desc = "XENOPHOBE_PRESAPIENTS_PURGE_DESC"
		
		fulfilled_effect = 10
		unfulfilled_effect = 0

		potential = { 
			exists = owner
			owner = { has_encountered_other_species = yes }
		}

		trigger = {
			owner = { has_policy_flag = pre_sapients_purge }
		}
	}
	
	# Wants Refugee Ban
	demand = {
		title = "XENOPHOBE_REFUGEE_BAN"
		unfulfilled_title = "XENOPHOBE_NO_REFUGEE_BAN"
		desc = "XENOPHOBE_REFUGEE_BAN_DESC"
		
		fulfilled_effect = 10
		unfulfilled_effect = 0

		potential = { 
			exists = owner
			owner = { has_encountered_other_species = yes }
		}

		trigger = {
			owner = { has_policy_flag = refugees_not_allowed }
		}
	}
	
	# Hates Migration Treaties
	demand = {
		title = "XENOPHOBE_MIGRATION_TREATIES"
		unfulfilled_title = "XENOPHOBE_NO_MIGRATION_TREATIES"
		desc = "XENOPHOBE_MIGRATION_TREATIES_DESC"

		fulfilled_effect = 0
		unfulfilled_effect = -20

		potential = {
			exists = owner
			owner = { num_communications > 2 }
		}

		trigger = {
			owner = {
				any_country = {
					is_country_type = default
					has_communications = root.owner
					NOR = {
						has_diplo_migration_treaty = root.owner
						root = {
							owner = {
								has_diplo_migration_treaty = prevprev
							}
						}
					}
				}
			}
		}
	}
	
	# Hates Alien Rights
	demand = {
		title = "XENOPHOBE_ALIEN_RIGHTS"
		unfulfilled_title = "XENOPHOBE_NO_ALIEN_RIGHTS"
		desc = "XENOPHOBE_ALIEN_RIGHTS_DESC"
		
		fulfilled_effect = 0
		unfulfilled_effect = -20

		potential = { 
			exists = owner
			owner = { has_encountered_other_species = yes }
		}

		trigger = {
			owner = { NOT = { has_policy_flag = alien_full_rights } }
		}
	}
	
	# Hates Protecting Presapients
	demand = {
		title = "XENOPHOBE_PRESAPIENT_PROTECT"
		unfulfilled_title = "XENOPHOBE_NO_PRESAPIENT_PROTECT"
		desc = "XENOPHOBE_PRESAPIENT_PROTECT_DESC"
		
		fulfilled_effect = 0
		unfulfilled_effect = -20

		potential = {
			exists = owner
			owner = { has_encountered_other_species = yes }
		}

		trigger = {
			owner = { NOT = { has_policy_flag = pre_sapients_protect } }
		}
	}
	
	# Hates No Pop Controls
	demand = {
		title = "XENOPHOBE_NO_POP_CONTROL"
		unfulfilled_title = "XENOPHOBE_NO_NO_POP_CONTROL"
		desc = "XENOPHOBE_NO_POP_CONTROL_DESC"
		
		fulfilled_effect = 0
		unfulfilled_effect = -10

		potential = { exists = owner }

		trigger = {
			owner = { NOT = { has_policy_flag = population_controls_not_allowed } }
		}
	}
	
	# Hates Embassies
	demand = {
		title = "XENOPHOBE_EMBASSIES"
		unfulfilled_title = "XENOPHOBE_NO_EMBASSIES"
		desc = "XENOPHOBE_NO_EMBASSIES_DESC"
		
		fulfilled_effect = 0
		unfulfilled_effect = -10

		potential = {
			exists = owner
			owner = { has_technology = tech_xenodiplomacy }
		}

		trigger = {
			owner = { NOT = { has_policy_flag = diplo_stance_cooperative } }
		}
	}
	
	# Hates Open Borders
	demand = {
		title = "XENOPHOBE_OPEN_BORDERS"
		unfulfilled_title = "XENOPHOBE_NO_OPEN_BORDERS"
		desc = "XENOPHOBE_OPEN_BORDERS_DESC"
		
		fulfilled_effect = 0
		unfulfilled_effect = -10

		potential = { exists = owner }

		trigger = {
			owner = { NOT = { has_policy_flag = border_policy_open } }
		}
	}
	
	# Hates Refugees
	demand = {
		title = "XENOPHOBE_REFUGEES_ALLOWED"
		unfulfilled_title = "XENOPHOBE_NO_REFUGEES_ALLOWED"
		desc = "XENOPHOBE_REFUGEES_ALLOWED_DESC"
		
		fulfilled_effect = 0
		unfulfilled_effect = -10

		potential = { 
			exists = owner
			owner = { has_encountered_other_species = yes }
		}

		trigger = {
			owner = { NOT = { has_policy_flag = refugees_allowed } }
		}
	}
	
	# OUTRAGE
	demand = {
		title = "FACTION_OUTRAGED_GC"
		unfulfilled_title = "FACTION_OUTRAGED_GC"
		desc = "FACTION_OUTRAGED_GC_DESC"

		fulfilled_effect = -20
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				has_country_flag = faction_xenophobe_supremacist_outraged_gc
			}
		}

		trigger = {
			owner = {
				has_country_flag = faction_xenophobe_supremacist_outraged_gc
			}
		}
	}
	
	on_create = {
		pop_faction_event = { id = factions.1 }
	}
	
	on_destroy = {
		if = {
			limit = {
				has_modifier = promoted_xenophobe
			}
			remove_modifier = promoted_xenophobe
		}
		if = {
			limit = {
				has_modifier = suppressed_xenophobe
			}
			remove_modifier = suppressed_xenophobe
		}
	}

	actions = {
		embrace_faction = {
			title = "EMBRACE_FACTION"
			description = "EMBRACE_FACTION_DESC"
			
			cost = {
				influence = 250
			}
			
			potential = {
				exists = owner
				owner = {
					OR = {
						is_subject = no
						NOT = { is_subject_type = dominion }
					}
				}
			}
			
			valid = {
				custom_tooltip = {
					fail_text = EMBRACE_FACTION_COOLDOWN
					parameter:empire = {
						NOT = { has_modifier = embraced_faction_timer }
					}
				}
				support > 0.20
				parameter:empire = {
					NOT = { has_ethic = "ethic_fanatic_xenophobe" }
				}
			}

			effect = {
				add_modifier = { modifier = embraced_faction days = 3600 }
				parameter:empire = {
					shift_ethic = ethic_xenophobe
					hidden_effect = {
						every_pop_faction = { # embraces xenophobe ethos
							limit = { is_pop_faction_type = isolationist }
							add_modifier = { modifier = embraced_faction days = 3600 }
						}
						add_modifier = { modifier = embraced_faction_timer days = 3600 }
						every_pop_faction = {
							limit = { NOT = { is_same_value = root } }
							add_modifier = { modifier = embraced_another_faction days = 3600 }
						}
					}
				}
				hidden_effect = {
					save_event_target_as = TargetFaction
					parameter:empire = {
						every_relation = {
							limit = {
								is_ai = no
								is_country_type = default			
								has_communications = prev
							}
							country_event = { id = factions.2000 }
						}
					}				
				}				
			}
			
			ai_weight = {
				base = 1
				modifier = {
					factor = 0
					support < 0.50
					owner = {
						has_ethic = ethic_xenophobe
					}
				}
				modifier = {
					factor = 0
					owner = { has_valid_civic = civic_fanatic_purifiers }
				}				
				modifier = {
					factor = 0
					NOT = { 
						owner = {
							exists = ruler
							ruler = {
								leader_of_faction = isolationist
							}
						}
					}
				}
				modifier = {
					factor = 0
					owner = {
						count_pop_factions = {
							count < 4
						}					
					}
				}				
			}
		}
		promote_faction = {
			title = "PROMOTE_FACTION"
			description = "PROMOTE_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_xenophobe
						has_modifier = promoted_xenophobe
						has_valid_civic = civic_anarchist
					}
				}
			}

			effect = {
				parameter:empire = {
					add_modifier = { modifier = promoted_xenophobe days = -1 }
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_promote_faction = {
			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = promoted_xenophobe
				}
			}

			effect = {
				parameter:empire = {
					remove_modifier = promoted_xenophobe
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_xenophobe
						has_modifier = promoted_xenophobe
						has_valid_civic = civic_anarchist
					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }
				parameter:empire = {
					hidden_effect = {
						every_pop_faction = {
							limit = { is_pop_faction_type = isolationist }
							add_modifier = { modifier = suppressed_faction days = -1 }
						}
					}
					add_modifier = { modifier = suppressed_xenophobe days = -1 }
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = suppressed_xenophobe
				}
			}

			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_xenophobe
					hidden_effect = {
						every_pop_faction = {
							limit = { is_pop_faction_type = isolationist }
							remove_modifier = suppressed_faction
						}
					}
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
	}
