energy_expenditure_colonies = {
	resource = energy 
	type = expenditure
	category = colonies
	
	potential = {
		ai_colonize_plans > 0
		OR = {
			has_civic = civic_corporate_dominion
			has_civic = civic_private_prospectors
		}
		has_resource = { type = energy amount > 1000 }
		has_monthly_income = { resource = food value > 0 }
	}
	
	static_min = {
		base = 500
	}
	static_max = {
		base = 500
	}
}
