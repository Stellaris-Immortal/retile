		#	}
		#}

			factor = 1.25
			has_trait = trait_charismatic
		}
		modifier = {
			factor = 0.9
			has_trait = trait_repugnant
		}	
		modifier = {
			factor = 0.1 # job is less useful if pop control is active, but still gives amenities

				has_modifier = planet_population_control_gestalt
			}
