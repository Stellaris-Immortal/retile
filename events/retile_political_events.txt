namespace = political

country_event = {
	id = political.1
	hide_window = yes

	immediate = {
		if = {
			limit = { check_variable = { which = unrest value > 100 } }
			set_variable = { which = unrest value = 100 }
		} else_if = {
			limit = { check_variable = { which = unrest value < 0 } }
			set_variable = { which = unrest value = 0 }
		}
	}
	
	trigger = {
		is_regular_empire = yes
		always = no
	}
	
	mean_time_to_happen = {
		days = 360
	}
	
	after = {
		random_list = {
			50 = { }
			# Authoritarian Request
			1 = { country_event = { id = political.2 } }
		}
	}
}

					limit = { is_pop_faction_type = totalitarian }
					add_modifier = { modifier = request_approved days = 3600 }
				}
			}
		} else_if = {
			limit = {
				has_country_flag = state_police_never
				has_country_flag = overlord_never
				has_country_flag = empire_never
			}
			hidden_effect = {
				remove_country_flag = state_police_never
				remove_country_flag = overlord_never
				remove_country_flag = empire_never
				every_pop_faction = {
					limit = { is_pop_faction_type = totalitarian }
					add_modifier = { modifier = request_approved days = 3600 }
				}
			}
		}
	}
	
	# You're Welcome

			has_policy_flag = criminal_state_police
			is_overlord = yes
			has_authority = auth_imperial
		}
		name = political.2.d
	}

# On action event that fires on ruler change 
# Function: if empire has dict/imp authority, fire additional events to simulate potential succession crisis
country_event = {
	id = political.101
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { has_authority = auth_dictatorial }
			country_event = { id = political.102 days = 5 }
		} else_if = {
			limit = { has_authority = auth_imperial }
			country_event = { id = political.103 days = 5 }
		}
	}
}

# Called by political.101
# Function: Create succession crisis for dictatorship
country_event = {
	id = political.102
	title = political.102.name
	desc = political.102.desc
	is_triggered_only = yes
	picture = GFX_evt_alien_segregation
	show_sound = event_public_unrest
	
	immediate = {
		every_planet_within_border = {
			limit = {
				is_colony = yes
			}
			add_modifier = { days = 1800 modifier = "retile_modifier_dict_succession_crisis" }
		}
	}
	
	option = {
		name = political.102.a
		tooltip = {
			add_modifier = {
				modifier = retile_modifier_dict_succession_crisis
				days = 1800
			}
		}
	}
}
